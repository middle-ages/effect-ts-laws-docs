Interface ParameterizedMap<F, A, B, C, In1, Out2, Out1>

Use module augmentation to add entries here for a new typeclasses. Used to map from typeclass name to its various test type. For example to get the types related to the Monad when testing the Option datatype with underlying values number/string/boolean:

type MyTypes = ParameterizedMap<OptionTypeLambda, number, string, boolean>['Monad']
// MyTypes ≡{
//   lambda: MonadTypeLambda
//   options: MonadOptions<OptionTypeLambda, number, string, boolean>
//   laws: LawList<[[a: number], [fa: Option<number>], [fa: Option<number>]]>
// }
interface ParameterizedMap<F, A, B, C, In1, Out2, Out1> {
    Applicative: {
        lambda: ApplicativeTypeLambda;
        laws: LawList<[[a: Kind<F, In1, Out2, Out1, A>], [a: A, fab: ((a: A) => B)], [fa: Kind<F, In1, Out2, Out1, A>, Fab: Kind<F, In1, Out2, Out1, ((a: A) => B)>, Fbc: Kind<F, In1, Out2, Out1, ((a: B) => C)>], [a: A, fab: Kind<F, In1, Out2, Out1, ((a: A) => B)>], [fa: Kind<F, In1, Out2, Out1, A>, ab: ((a: A) => B)]]>;
        options: CommonOptions<ApplicativeTypeLambda, F, A, B, C, In1, Out2, Out1>;
    };
    Covariant: {
        lambda: CovariantTypeLambda;
        laws: LawList<[[a: Kind<F, In1, Out2, Out1, A>], [a: Kind<F, In1, Out2, Out1, A>, fab: ((a: A) => B), fbc: ((a: B) => C)]]>;
        options: CommonOptions<CovariantTypeLambda, F, A, B, C, In1, Out2, Out1>;
    };
    Invariant: {
        lambda: InvariantTypeLambda;
        laws: LawList<[[a: Kind<F, In1, Out2, Out1, A>], [a: Kind<F, In1, Out2, Out1, A>, fab: ((a: A) => B), fbc: ((b: B) => C), fba: ((b: B) => A), fcb: ((c: C) => B)]]>;
        options: CommonOptions<InvariantTypeLambda, F, A, B, C, In1, Out2, Out1>;
    };
    Monad: {
        lambda: MonadTypeLambda;
        laws: LawList<[[a: A, faB: ((a: A) => Kind<F, In1, Out2, Out1, B>)], [fa: Kind<F, In1, Out2, Out1, A>], [fa: Kind<F, In1, Out2, Out1, A>, faB: ((a: A) => Kind<F, In1, Out2, Out1, B>), fbC: ((b: B) => Kind<F, In1, Out2, Out1, C>)], [fa: Kind<F, In1, Out2, Out1, A>, ab: ((a: A) => B)]]>;
        options: MonadOptions<F, A, B, C, In1, Out2, Out1>;
    };
}
